" Function {{{

" Remove trailing whitespace when writing a buffer, but not for diff files
function! RemoveTrailingWhitespace()
  if &ft != "diff"
    let b:curcol = col(".")
    let b:curline = line(".")
    silent! %s/\s\+$//
    silent! %s/\(\s*\n\)\+\%$//
    call cursor(b:curline, b:curcol)
  endif
endfunction

" Return to last edit line
function! BackToLastEditingLine()
  if line("'\"") > 1 && line("'\"") <= line("$") |
    exe "normal! g`\"" |
  endif
endfunction

" ----- Call Function ----- {{{

" global
au BufWritePre * call RemoveTrailingWhitespace()
au BufReadPost * call BackToLastEditingLine()

" filetype vim
augroup ft_vim
  au!

  au FileType vim setlocal foldlevel=0
  au FileType vim setlocal foldmethod=marker
augroup END
" }}}
