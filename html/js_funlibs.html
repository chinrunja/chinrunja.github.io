	// 输出num范围内的素数（质数）
	function showPrimes(num) {

		for (let i = 2; i < num; i++) {
			if ( isPrime(i) ) {
				console.log(i);
			}
		}

	}

	function isPrime(num) {
	    
		for (let j = 2; j < num; j++) {
			if (num % j === 0) {
				return false;
			}
		}

		return true;
	}

	showPrime(10); // 2,3,5,7

3.递归返回范围内数组


	// 递归返回正序范围内数组
	function rangeOfNumbers(startNum, endNum) {

		if (startNum === endNum) return [startNum];
		else if (startNum < endNum) {
			endNum--;
			return ( rangeOfNumbers(startNum, endNum) ).concate(endNum + 1);
		}
	}

	rangeOfNumbers(4,7); // [4,5,6,7]

	// 递归返回逆序范围内数组
	functionction rangeOfNumbers(startNum, endNum) {

		if (startNum === endNum) return [endNum];
		else if (startNum < endNum) {
			startNum++;
			return ( rangeOfNumbers(startNum, endNum) ).concate(startNum - 1);
		}
	}

	rangeOfNumbers(4,7); // [7,6,5,4]

4. 生成范围内随机数（包含起止点）


	// 生成范围内随机数（包含起止点）
	function randomInteger(min, max) {
    	return Math.floor(Math.random() * (max - min + 1)) + min;
	}

	randomInteger(1, 5);
