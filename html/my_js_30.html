    function setDate() {

第八天： HTML5 Canvas

查看我的完成示例

今天的示例是Canvas，但不是我学JS的重点，所以敷衍的完成了，从指南可以复制大部分代码，再把操作逻辑添加进去就好了


	// JS 代码
	const canvas = document.getElementById('draw');
	const ctx = canvas.getContext('2d');
	ctx.lineWidth = 100;
	canvas.width = window.innerWidth;
	canvas.height = window.innerHeight;

	let isDrawing = false,
		direction = true,
		hue = 0,
		lastX = 0,
		lastY = 0;

	function draw(e) {

		if (!isDrawing) return;

		ctx.beginPath();
		ctx.moveTo(lastX, lastY);
		ctx.lineTo(e.offsetX, e.offsetY);
		ctx.strokeStyle = `hsl(${ hue }, 100%, 50%)`;
		ctx.stroke();
		[lastX, lastY] = [e.offsetX, e.offsetY];

		if(hue >= 360) hue = 0;
		hue++;

		if(ctx.lineWidth >= 100 || ctx.lineWidth <= 1) {
			direction = !direction;
		}

		if (direction) {
			ctx.lineWidth++;
		} else {
			ctx.lineWidth--;
		}
	}

	canvas.addEventListener('mousedown', (e) => {
		isDrawing = true;
		[lastX, lastY] = [e.offsetX, e.offsetY];
	});
	canvas.addEventListener('mousemove', draw);
	canvas.addEventListener('mouseup', (e) => {
		isDrawing = false;
	});
	canvas.addEventListener('mouseout', (e) => {
		isDrawing = false;
	});

第十三天： Slide in on Scroll

查看我的完成示例

指南已给出完整的实现方法，只需要遍历图片加入判断条件即可实现效果


	function debounce(func, wait = 20, immediate = true) {
      var timeout;
      return function() {
        var context = this, args = arguments;
        var later = function() {
          timeout = null;
          if (!immediate) func.apply(context, args);
        };
        var callNow = immediate && !timeout;
        clearTimeout(timeout);
        timeout = setTimeout(later, wait);
        if (callNow) func.apply(context, args);
      };
    }

    const slideImages = document.querySelectorAll('.slide-in');

    function checkSlide(e) {
      console.log(e);
      console.count(e);
      
      slideImages.forEach(img => {
        const slideInAt = (window.scrollY + window.innerHeight) - img.height / 2;
        const imgBottom = img.offsetTop + img.height;
        const isHalfShow = slideInAt > img.offsetTop;
        const isNotScrollPast = window.scrollY < imgBottom;

        if (isHalfShow && isNotScrollPast) {
          img.classList.add('active');
        } else {
          img.classList.remove('active');
        }
      });
    }
    
    window.addEventListener('scroll', debounce(checkSlide));
